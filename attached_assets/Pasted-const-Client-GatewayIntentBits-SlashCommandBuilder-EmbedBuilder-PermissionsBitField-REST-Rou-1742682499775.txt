  const messageId = reaction.message.id;
  const poll = config.activePolls.get(messageId);
  if (!poll) return;

  const emoji = reaction.emoji.name;
  const optionIndex = reactionEmojis.indexOf(emoji);
  if (optionIndex === -1) return;
